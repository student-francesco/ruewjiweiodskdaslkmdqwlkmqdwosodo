const getCoords = (r: number, radians: number) => [r * Math.cos(radians), r * Math.sin(radians)];

const r = 75;
const angle = 45 * Math.PI / 180;
const angleOffset = 30 * Math.PI / 180
const coords_a = getCoords(r, angle - angleOffset).map(v => v * -1 + 'px').join(',')
const coords_b = getCoords(r, angle).map(v => v * -1 + 'px').join(',')
const coords_c = getCoords(r, angle + angleOffset).map(v => v * -1 + 'px').join(',')

const r_sm = 69;
